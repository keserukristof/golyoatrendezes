3   import java.beans.XMLDecoder;

6   import java.util.ArrayList;

8   public class Ranglista {
9   
10      private ArrayList<Rekord> rangsor = new ArrayList<>();
11  
12      public Ranglista() {}
13  
14      /**
15       * @param nev a felhasználó neve
16       * @param lepes a játékban megtett lépésszám
17       * Ha a rangsor mérete nagyobb lesz mint 10 akkor az utolsót eltávolítja
18       *
19       */
20      public void beszurHaTud(String nev, int lepes) {
21          RekordRekordd = new Rekord();
22          rekord.setNev(nev);
23          rekord.setLepes(lepes);
24          rangsor.add(rekord);
25          rangsor.sort(new RekordKomparator());
26          if (rangsor.size() > 10) {
27              rangsor.remove(10);
28          }
29  
30      }
31  
32      /**
33       * @return Stringé alakított ret érték
34       */
35      @Override
36      public String toString() {
37          String ret = "";
38          for (Rekord r : rangsor) {
39              ret += r.getNev() + " " + r.getLepes() + "\n";
40          }
41          return ret;
42      }
43  
44      public void listaz() {
45          System.out.println(toString());
46      }
47  
48  
49      /**
50       * @param file a fájl nevét várja paraméterül amibe ki fog írni
51       */
52      public void kiir(String file) {
53          XMLEncoder encoder=null;
54          try {
55              encoder=new XMLEncoder(new BufferedOutputStream(new FileOutputStream(file)));
56              encoder.writeObject(rangsor);
57              encoder.close();
58          } catch(FileNotFoundException fileNotFound){
59              Logger.error("ERROR: While Creating or Opening the File " + file);
60          }
61  
62      }
63  
64  
65      /**
66       * @param file a fájl nevét kéri paraméterül amiből beolvassa a ranglistát
67       */
68      public void beolvas(String file) {
69          XMLDecoder decoder = null;
70          try {
71              decoder = new XMLDecoder(new BufferedInputStream(new FileInputStream(file)));
72              rangsor = (ArrayList<Rekord>)decoder.readObject();;
73          } catch (FileNotFoundException e) {
74              Logger.error("ERROR: File " + file + " not found");
75          }
76      }
